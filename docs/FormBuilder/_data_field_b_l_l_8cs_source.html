344 public string GetDisplayValue(VM.DataField dataField)
345 {
346 var displayValue = String.Empty;
347
348 if (string.IsNullOrWhiteSpace(dataField.Value))
349 return string.Empty;
350
351 switch ((Enums.FormEnums.FieldType) dataField.Field.FieldTypeId)
352 {
353 case Enums.FormEnums.FieldType.Input:
354 case Enums.FormEnums.FieldType.TextArea:
355 case Enums.FormEnums.FieldType.InputNumber:
356 case Enums.FormEnums.FieldType.TextAreaByEditor:
357 case Enums.FormEnums.FieldType.DatePicker:
358 case Enums.FormEnums.FieldType.TimePicker:
359 //Exactly Value
360 displayValue = dataField.Value;
361
362 break;
363
364 case Enums.FormEnums.FieldType.Select:
365 case Enums.FormEnums.FieldType.RadioBTN:
366 //check zero Value
367 if (dataField.Value == "0")
368 return string.Empty;
369
370 displayValue = ItemsBLL.Instance.GetById(Int32.Parse(dataField.Value)).NameFa;
371
372 break;
373
374 case Enums.FormEnums.FieldType.MultiSelect:
375 var itemIds = dataField.Value.Split(',');
376
377 var counter = itemIds.Length;
378
379 foreach (var itemId in itemIds)
380 {
381 var intId = Int32.Parse(itemId);
382
383 if (intId > 0)
384 displayValue += ItemsBLL.Instance.GetById(intId).NameFa;
385
386 counter--;
387
388 if (counter > 1)
389 displayValue += "-";
390
391 }
392
393 break;
394 }
395
396 return displayValue;
397 }
